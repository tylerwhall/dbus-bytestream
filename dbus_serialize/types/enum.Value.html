Identity function for Value which unpacks Variants

Allows the use of a Value and a Variant of a Value interchangeably.

Examples

use dbus_serialize::types::{Value, Variant};

let val = Value::from(12);
let variant = Value::Variant(Variant::new(Value::from(12), "i"));
assert_eq!(val.value(), variant.value());
